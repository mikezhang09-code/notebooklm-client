#
#   python3.12 -m venv ~/oml-venv
#   ~/oml-venv/bin/pip install --upgrade pip
#   # CPU-only torch (smaller, no GPU needed for a one-time export):
#   ~/oml-venv/bin/pip install torch==2.6.0 --index-url https://download.pytorch.org/whl/cpu
#   ~/oml-venv/bin/pip install -r requirements.txt
#   # then the OML4Py client wheel (NOT on PyPI — see README.md):
#   ~/oml-venv/bin/pip install oml-2.1.1-cp312-cp312-linux_x86_64.whl
#
# NOTE: torch is intentionally NOT pinned here — install it separately from the
# CPU index above so pip doesn't pull the multi-GB CUDA build.
